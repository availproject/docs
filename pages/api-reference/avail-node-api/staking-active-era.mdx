## Minimal example (Fetch the active era)

<Callout type="info">

import { SDK } from "avail-js-sdk"

const main = async () => {
  const providerEndpoint = "wss://turing-rpc.avail.so/ws"
  const sdk = await SDK.New(providerEndpoint)

<Tabs.Tab>

1. Inside `main.rs`, add the following code:

```rust filename="avail-rust" filename="avail-rust"
use avail_rust::{avail, SDK};
 
#[tokio::main]
async fn main() -> Result<(), String> {
	let sdk = SDK::new("wss://turing-rpc.avail.so/ws").await.unwrap();
 
	let storage_query = avail::storage().staking().active_era();
	let best_block_hash = sdk
		.rpc
		.chain
		.get_block_hash(None)
		.await
		.map_err(|e| e.to_string())?;
	let result = sdk
		.api
		.storage()
		.at(best_block_hash)
		.fetch(&storage_query)
		.await
		.map_err(|e| e.to_string())?;
 
	dbg!(result);
 
	Ok(())
}
```

2. Run the code using:

```bash
cargo run
```
